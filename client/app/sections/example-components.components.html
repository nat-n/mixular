    A radio set component

    angular.module('mixularApp')

  .component('mxRadioSet', {
    core: true,
    controller: function() {
      this.options = [];
    },
    link: {
      pre: function (scope, elem, attrs, ctrl) {
        ctrl.label = attrs.label;
      }
    },
    template: (
      '<fieldset>' +
        '<legend ng-show="mx.label" ng-bind="mx.label"></legend>' +
        '<div>' +
          '<label class="radio" ng-repeat="o in mx.options">' +
            '<input type="radio"' +
                  ' ng-model="mx.model[mx.key]"' +
                  ' name="{{mx.id}}"' +
                  ' id="{{mx.id}}-{{$index}}"' +
                  ' ng-value="o.value">' +
            '{{o.label}}' +
          '</label>' +
        '</div>' +
      '</fieldset>'
    ),
    targets: function(elem) {
      var fieldset = elem.children('fieldset')[0];
      return {
        main: fieldset,
        legend: elem.find('legend')[0],
        labelsWrapper: elem.find('fieldset > div')[0]
      };
    }
  });

  • The mx-options mixin automatically works on the radioset component, because it also works by maintaining an array of options on its controller.
  • The mx-options-list mixin is interchangable with the mx-options mixin, and works instead by referencing a list of options maintained by a service (which in this case is updated by a watcher which whatches the model of the other fields).
